json的规格：

  1. 并列的数据之间用逗号(",")分隔
  2. 映射用冒号(":")表示
  3. 并列数据的集合(数组)用方括号("[]")表示
  4. 映射的集合(对象)用大括号("{}")表示

json解析的例子（获取name）：

  1. json1={"name":"Kaitlyn","tel":"18912345678"}

    json1.name或者json1["name"]

  2. json2=[{"name":"Kaitlyn","tel":"18912345678"},{"name":"Kaitlyn1","tel":"18987654321"},{"name":"Kaitlyn2","tel":"18998765432"}]

    json2[0].name或者json2[0]["name"]

  3. json3="{'name':'Kaitlyn','tel':'18912345678'}"

    var data=eval('('+json3+')');data["name"]

  4. json4="[{'name':'Kaitlyn','tel':'18912345678'},{'name':'Kaitlyn1','tel':'18987654321'},{'name':'Kaitlyn2','tel':'18998765432'}]"

    var data=eval(json4);data[0]["name"]
